<Story name="basic">
  <div>
    <List.Accordion title="List Accordion" startDeg="90deg" expanded={false}>
      <div style={{margin: 8}}>
        {Array(3)
          .fill('')
          .map((i, k) => (
            <List.Item
              key={k}
              type="avatar"
              left={({...props}) => (
                <IconButton {...props} icon="person" size={24} />
              )}
              right={({...props}) => (
                <IconButton {...props} icon="chevron_right" />
              )}
              title={'item ' + (k + 4)}
              subtitle={'Subtitle ' + (k + 4)}
            />
          ))}
      </div>
    </List.Accordion>
  </div>
</Story>
